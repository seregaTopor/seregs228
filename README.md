//

    Если Ссылка.Пустая() Тогда

    КонецЕсли;

        ВызватьИсключение "Ошибка: Партнер не найден";
    КонецЕсли;

        Партнер.Наименование = СокрЛП(Наименование);
    КонецЕсли;

    КонецЕсли;

    КонецЕсли;

        Партнер.Email = СокрЛП(Email);
    КонецЕсли;
    
    Если НЕ ПустаяСтрока(Адрес) Тогда
        Партнер.Адрес = СокрЛП(Адрес);
    КонецЕсли;

        Партнер.Записать();
        ЗаписатьВЖурналРегистрации("Обновлен партнер: " + Партнер.Наименование);
        Возврат Истина;
    Исключение
        ВызватьИсключение "Ошибка при обновлении партнера: " + ОписаниеОшибки();
    КонецПопытки;

//
// Параметры:

//   ЛогическоеУдаление - Булево - Истина - установить статус "Удален", Ложь - физическое удаление
//
// Возвращаемое значение:
//   Булево - Истина если удаление успешно
//
Функция УдалитьПартнера(Ссылка, ЛогическоеУдаление = Истина)
    
    Если Ссылка.Пустая() Тогда
        ВызватьИсключение "Ошибка: Ссылка на партнера не может быть пустой";
    КонецЕсли;
    
    Если ЛогическоеУдаление Тогда
        // Логическое удаление - меняем статус
        Партнер = Ссылка.ПолучитьОбъект();
        Партнер.Статус = Перечисления.СтатусыПартнеров.Удален;
        Партнер.ДатаУдаления = ТекущаяДата();
        Партнер.Записать();
        ЗаписатьВЖурналРегистрации("Логически удален партнер: " + Партнер.Наименование);
    Иначе
        // Физическое удаление
        Попытка
            Ссылка.Удалить();
            ЗаписатьВЖурналРегистрации("Физически удален партнер: " + Ссылка);
        Исключение
            ВызватьИсключение "Ошибка при удалении партнера: " + ОписаниеОшибки();
        КонецПопытки;
    КонецЕсли;
    
    Возврат Истина;

#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""
Скрипт импорта данных из JSON файлов в Redis
День 6: Ubuntu Server + Redis + API
Дата: 28.04.2026
"""

import json
import redis
import os
import sys

# Конфигурация подключения к Redis
REDIS_HOST = 'localhost'
REDIS_PORT = 6379
REDIS_DB = 0

def connect_redis():
    """Подключение к Redis"""
    try:
        r = redis.Redis(
            host=REDIS_HOST,
            port=REDIS_PORT,
            db=REDIS_DB,
            decode_responses=True
        )
        r.ping()
        print("✅ Подключение к Redis успешно установлено")
        return r
    except Exception as e:
        print(f"❌ Ошибка подключения к Redis: {e}")
        sys.exit(1)

def clear_redis(r):
    """Очистка Redis (опционально)"""
    response = input("Очистить Redis перед импортом? (y/n): ")
    if response.lower() == 'y':
        r.flushdb()
        print("🗑️ Redis очищен")
    else:
        print("📦 Данные в Redis сохранены")

def import_users(r, filepath):
    """Импорт пользователей из JSON"""
    print("\n📖 Импорт пользователей...")
    
    with open(filepath, 'r', encoding='utf-8') as f:
        users = json.load(f)
    
    for user in users:
        user_id = user['_id']['$oid']
        
        # Преобразуем дату
        if 'created_at' in user and '$date' in user['created_at']:
            user['created_at'] = user['created_at']['$date']
        
        # Сохраняем как JSON
        r.set(f'user:{user_id}', json.dumps(user, ensure_ascii=False))
        r.sadd('users:list', user_id)
        r.set(f'user:username:{user["username"]}', user_id)
        r.set(f'user:email:{user["email"]}', user_id)
        
        # Сортированный набор по возрасту
        r.zadd('users:by_age', {user_id: user['age']})
        
        # Индекс по городу
        city = user['address']['city']
        r.sadd(f'city:{city}', user_id)
        
        print(f"  ✓ {user['username']} (ID: {user_id}, возраст: {user['age']})")
    
    return len(users)

def import_products(r, filepath):
    """Импорт товаров из JSON"""
    print("\n📦 Импорт товаров...")
    
    with open(filepath, 'r', encoding='utf-8') as f:
        products = json.load(f)
    
    for product in products:
        product_id = product['_id']['$oid']
        
        # Сохраняем как JSON
        r.set(f'product:{product_id}', json.dumps(product, ensure_ascii=False))
        r.sadd('products:list', product_id)
        r.sadd(f'category:{product["category"]}', product_id)
        
        # Индекс по цене (сортированный набор)
        r.zadd('products:by_price', {product_id: product['price']})
        
        # Индекс по наличию
        if product.get('in_stock', False):
            r.sadd('products:in_stock', product_id)
        else:
            r.sadd('products:out_of_stock', product_id)
        
        # Индекс по тегам
        for tag in product.get('tags', []):
            r.sadd(f'tag:{tag}', product_id)
        
        print(f"  ✓ {product['name']} (ID: {product_id}, цена: {product['price']}₽)")
    
    return len(products)

def print_statistics(r):
    """Вывод статистики"""
    print("\n" + "="*50)
    print("📊 СТАТИСТИКА ИМПОРТА")
    print("="*50)
    print(f"👥 Пользователей: {r.scard('users:list')}")
    print(f"📦 Товаров: {r.scard('products:list')}")
    
    # Возрастная статистика
    print("\n📈 Возрастное распределение:")
    ages = r.zrange('users:by_age', 0, -1, withscores=True)
    for user_id, age in ages:
        username = r.get(f'user:username:{user_id}')
        if username:
            print(f"  - {username}: {int(age)} лет")
    
    # Статистика по категориям
    print("\n🏷️ Категории товаров:")
    categories = ['Электроника', 'Спорт и отдых', 'Книги']
    for cat in categories:
        count = r.scard(f'category:{cat}')
        print(f"  - {cat}: {count} товаров")
    
    print("\n✅ Импорт успешно завершён!")

        # Возрастная статистика пользователей
        user_ages = r.zrange('users:by_age', 0, -1, withscores=True)
        ages = [int(age) for _, age in user_ages]
        if ages:
            stats['user_age_stats'] = {
                "min": min(ages),

@app.get("/synthesize")
async def synthesize_speech(
    text: str = Query(..., description="Текст для озвучивания", min_length=1, max_length=500),
    voice: str = Query("irina", description="Голос (irina - русский)"),
    format: str = Query("wav", description="Формат аудио (wav)")
):
    """
    Синтез речи из текста
    Возвращает аудиофайл в формате WAV
    """
    if not text:
        raise HTTPException(status_code=400, detail="Параметр 'text' обязателен")
    
    # Проверка модели
    ok, msg = check_model_exists()
    if not ok:
        raise HTTPException(status_code=503, detail=f"TTS модель не доступна: {msg}")
    
    # Генерация уникального имени файла
    filename = f"{uuid.uuid4()}.{format}"
    output_file = os.path.join(OUTPUT_DIR, filename)
    
    # Экранирование текста для shell
    safe_text = text.replace('"', '\\"').replace('$', '\\$').replace('`', '\\`')
    
    # Формирование команды
    cmd = f'echo "{safe_text}" | {PIPER_BIN} --model {TTS_MODEL_PATH} --output_file {output_file}'
    
    try:
        # Запуск Piper TTS
        result = subprocess.run(
            cmd,
            shell=True,
            capture_output=True,
            text=True,
            timeout=30
        )
        
        if result.returncode != 0:
            error_msg = result.stderr.strip() if result.stderr else "Неизвестная ошибка"
            raise HTTPException(
                status_code=500,
                detail=f"Ошибка синтеза речи: {error_msg}"
            )
        
        # Проверка, что файл создан
        if not os.path.exists(output_file) or os.path.getsize(output_file) == 0:
            raise HTTPException(status_code=500, detail="Аудиофайл не создан или пуст")
        
        # Возврат файла
        return FileResponse(
            output_file,
            media_type="audio/wav",
            filename=f"speech_{uuid.uuid4()}.wav",
            headers={
                "Content-Disposition": f"attachment; filename=speech_{uuid.uuid4()}.wav"
            }
        )
        
    except subprocess.TimeoutExpired:
        raise HTTPException(status_code=504, detail="Превышено время ожидания синтеза")
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))

@app.post("/synthesize/batch")
async def synthesize_batch(texts: List[str] = Query(...)):
    """
    Пакетный синтез речи
    Возвращает архив с аудиофайлами
    """
    if not texts:
        raise HTTPException(status_code=400, detail="Список текстов пуст")
    
    if len(texts) > 10:
        raise HTTPException(status_code=400, detail="Максимум 10 текстов за раз")
    
    results = []
    for i, text in enumerate(texts):
        try:
            # Генерация уникального имени файла
            filename = f"{uuid.uuid4()}.wav"
            output_file = os.path.join(OUTPUT_DIR, filename)
            
            safe_text = text.replace('"', '\\"').replace('$', '\\$').replace('`', '\\`')
            cmd = f'echo "{safe_text}" | {PIPER_BIN} --model {TTS_MODEL_PATH} --output_file {output_file}'
            
            result = subprocess.run(cmd, shell=True, capture_output=True, text=True, timeout=30)
            
            if result.returncode == 0 and os.path.exists(output_file):
                results.append({
                    "index": i,
                    "text": text[:50] + "..." if len(text) > 50 else text,
                    "status": "success",
                    "file": filename
                })
            else:
                results.append({
                    "index": i,
                    "text": text[:50] + "..." if len(text) > 50 else text,
                    "status": "failed",
                    "error": result.stderr.strip() if result.stderr else "Неизвестная ошибка"
                })
        except Exception as e:
            results.append({
                "index": i,
                "text": text[:50] + "..." if len(text) > 50 else text,
                "status": "failed",
                "error": str(e)
            })
    
    return {
        "total": len(texts),
        "successful": len([r for r in results if r["status"] == "success"]),
        "failed": len([r for r in results if r["status"] == "failed"]),
        "results": results
    }
